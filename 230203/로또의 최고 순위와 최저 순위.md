```python
# Q1 Answer template

def solution(lottos, win_nums):
    
    zero_cnt = 0
    collect_cnt = 0
    answer = []
    
    for l in lottos:
        if l in win_nums:
            collect_cnt += 1
        elif l == 0:
            zero_cnt += 1
    
    cnt = [6, 6, 5, 4, 3, 2, 1]
    
    answer = [cnt[collect_cnt+zero_cnt], cnt[collect_cnt]]
    
    return answer
    ```
